Процедура Расш1_ОтправитьКартыПередНаСервере()



  БазовыйURLСайта = "https://metaphysica-parfums.com";


  ТокенСайта = "nhjbjhmbkbk";


  


  МассивКарт = Новый Массив;


  


  Запрос = Новый Запрос;


  Запрос.Текст = 


    "ВЫБРАТЬ


    |  ПодарочныеКарты.Ссылка.Код КАК external_id,


    |  ПодарочныеКарты.card_number КАК card_number,


    |  ПодарочныеКарты.nominal КАК nominal,


    |  ПодарочныеКарты.status_1c КАК status_1c


    |ИЗ


    |  Справочник.ПодарочныеКарты КАК ПодарочныеКарты";


  


  РезультатЗапроса = Запрос.Выполнить();


  Выборка = РезультатЗапроса.Выбрать();


  


  Пока Выборка.Следующий() Цикл


    


    Карта = Новый Структура;


    Карта.Вставить("external_id", Строка(Выборка.external_id));


    Карта.Вставить("card_number", Строка(Выборка.card_number));


    


    Если Выборка.nominal <> Неопределено Тогда


      Карта.Вставить("nominal", Выборка.nominal);


    Иначе


      Карта.Вставить("nominal", Неопределено);


    КонецЕсли;


    


    Карта.Вставить("status_1c", Строка(Выборка.status_1c));


    МассивКарт.Добавить(Карта);


    


  КонецЦикла;


  


  Если МассивКарт.Количество() = 0 Тогда


    Сообщить("Нет карт для отправки!");


    Возврат;


  КонецЕсли;


  


  СтруктураДляJSON = Новый Структура;


  СтруктураДляJSON.Вставить("cards", МассивКарт);


  


  ЗаписьJSON = Новый ЗаписьJSON;


  ЗаписьJSON.УстановитьСтроку();


  ЗаписатьJSON(ЗаписьJSON, СтруктураДляJSON);


  JSONТекст = ЗаписьJSON.Закрыть();


  


ПутьЗапроса = "/wp-json/wgpc/v1/cards/import";


  


  HTTPСоединение = Новый HTTPСоединение(


    "metaphysica-parfums.com",


    443,


    ,


    ,


    ,


    ,


    Новый ЗащищенноеСоединениеOpenSSL()


  );


  


  HTTPЗапрос = Новый HTTPЗапрос(ПутьЗапроса);


  HTTPЗапрос.Заголовки.Вставить("Content-Type", "application/json; charset=utf-8");


  HTTPЗапрос.Заголовки.Вставить("X-WGPC-Token", ТокенСайта);


  HTTPЗапрос.УстановитьТелоИзСтроки(JSONТекст, КодировкаТекста.UTF8);


  


  Попытка


    HTTPОтвет = HTTPСоединение.ВызватьHTTPМетод("POST", HTTPЗапрос);    


    Если HTTPОтвет.КодСостояния = 200 ИЛИ HTTPОтвет.КодСостояния = 207 Тогда


      ТекстОтвета = HTTPОтвет.ПолучитьТелоКакСтроку(КодировкаТекста.UTF8);


      Сообщить("Успешно отправлено! Карт: " + МассивКарт.Количество());


      Сообщить("Ответ сервера: " + ТекстОтвета);


    Иначе


      Сообщить("Ошибка! Код ответа: " + HTTPОтвет.КодСостояния);


      ТекстОтвета = HTTPОтвет.ПолучитьТелоКакСтроку(КодировкаТекста.UTF8);


      Сообщить("Ответ: " + ТекстОтвета);


    КонецЕсли;


    


  Исключение


    Сообщить("Ошибка при отправке: " + ОписаниеОшибки());


  КонецПопытки;

Функция Расш1_ПолучитьСтатусыСайтаПередНаСервере()


    


  ОбработанныеExternalID = Новый Массив;    


  ТокенСайта = "nhjbjhmbkbk";


  ПутьЗапроса = "/wp-json/wgpc/v1/cards/exports";


  


  HTTPСоединение = Новый HTTPСоединение(


    "metaphysica-parfums.com",


    443,


    ,


    ,


    ,


    ,


    Новый ЗащищенноеСоединениеOpenSSL()


  );


  


  HTTPЗапрос = Новый HTTPЗапрос(ПутьЗапроса);


  HTTPЗапрос.Заголовки.Вставить("X-WGPC-Token", ТокенСайта);


  


  Попытка


    HTTPОтвет = HTTPСоединение.Получить(HTTPЗапрос);


    ТекстОтвета = HTTPОтвет.ПолучитьТелоКакСтроку(КодировкаТекста.UTF8);


    


    Сообщить("Код ответа: " + HTTPОтвет.КодСостояния);


    


    Если HTTPОтвет.КодСостояния <> 200 Тогда


      Сообщить("Ответ сайта: " + ТекстОтвета);


      Возврат ОбработанныеExternalID;


    КонецЕсли;


    


    ЧтениеJSON = Новый ЧтениеJSON;


    ЧтениеJSON.УстановитьСтроку(ТекстОтвета);


    ДанныеJSON = ПрочитатьJSON(ЧтениеJSON);


    


    КартыСайта = Неопределено;


    Если Не ДанныеJSON.Свойство("cards", КартыСайта) Тогда


      Сообщить("В ответе сайта нет поля cards");


      Возврат ОбработанныеExternalID;


    КонецЕсли;


    


    КоличествоОбновленных = 0;


    


    Для Каждого КартаСайта Из КартыСайта Цикл


    КонецЦикла;


    


    Для Каждого КартаСайта Из КартыСайта Цикл


      


      ВнешнийИД = "";


      СтатусСайта = "";


      НомерЗаказа = "";


      ВалютаСайта = "";


      БалансСайта = Неопределено;


      


      Если КартаСайта.Свойство("external_id", ВнешнийИД) = Ложь Тогда


        Продолжить;


      КонецЕсли;


      


      Если КартаСайта.Свойство("status", СтатусСайта) = Ложь Тогда


        Продолжить;


      КонецЕсли;


      


      КартаСайта.Свойство("order_id", НомерЗаказа);


      КартаСайта.Свойство("currency_code", ВалютаСайта);


      КартаСайта.Свойство("balance", БалансСайта);


      


      СсылкаКарты = Справочники.ПодарочныеКарты.НайтиПоКоду(Строка(ВнешнийИД));


      Если СсылкаКарты.Пустая() Тогда


        Сообщить("Карта в 1С не найдена по коду: " + Строка(ВнешнийИД));


        Продолжить;


      КонецЕсли;


      


      ОбъектКарты = СсылкаКарты.ПолучитьОбъект();


      


      Если СтатусСайта = "activated" Тогда


        ОбъектКарты.status_1c = "Активирована";


      ИначеЕсли СтатусСайта = "sold" Тогда


        ОбъектКарты.status_1c = "Продана";


      Иначе


        ОбъектКарты.status_1c = Строка(СтатусСайта);


      КонецЕсли;


      


      Если НомерЗаказа <> Неопределено И НомерЗаказа <> Null Тогда


        ОбъектКарты.order_id = Строка(НомерЗаказа);


      КонецЕсли;


      Если ВалютаСайта <> Неопределено И ВалютаСайта <> Null И Строка(ВалютаСайта) <> "" Тогда


        Расш1_УстановитьРеквизитЕслиЕсть(ОбъектКарты, "currency_code", Строка(ВалютаСайта));


      КонецЕсли;


      Если БалансСайта <> Неопределено И БалансСайта <> Null Тогда


        Расш1_УстановитьРеквизитЕслиЕсть(ОбъектКарты, "balance", БалансСайта);


      КонецЕсли;


      


      ОбъектКарты.Записать();        


      ОбработанныеExternalID.Добавить(Строка(ВнешнийИД));


      КоличествоОбновленных = КоличествоОбновленных + 1;


      


    КонецЦикла;


    


    Сообщить("Обновлено карт в 1С: " + КоличествоОбновленных);


    


  Исключение


    Сообщить("Ошибка при получении статусов: " + ОписаниеОшибки());


  КонецПопытки;



    Возврат ОбработанныеExternalID;
